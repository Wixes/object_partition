var scene, camera, renderer, controls;
var spotlight;

scene = new THREE.Scene();
scene.background = new THREE.Color({color: 0x00ff00});

camera = new THREE.PerspectiveCamera(90, window.innerWidth / window.innerHeight, 0.1, 1000);
camera.position.set(0, 1, 2);
camera.lookAt(scene.position);

renderer = new THREE.WebGLRenderer({canvas: artifactCanvas,
                                    alpha: true,
                                    antialias: true});


// CUBE
var cubeGeometry = new THREE.CubeGeometry(1,1,1);
var cubeMaterial = new THREE.MeshLambertMaterial({color: 0xff0000});
var cube = new THREE.Mesh(cubeGeometry, cubeMaterial);

// SPHERE
var sphereGeometry = new THREE.SphereGeometry(1, 32, 32);
var sphereMaterial = new THREE.MeshLambertMaterial({color: 0x00ff00});
var sphere = new THREE.Mesh(sphereGeometry, sphereMaterial);

//LIGHT
spotlight_1 = new THREE.SpotLight(0xffffff);
spotlight_2 = new THREE.SpotLight(0xffffff);

// CONTROLLER
controls = new THREE.OrbitControls( camera, renderer.domElement );
controls.rotateSpeed = 0.1;
controls.zoomSpeed = 1.0;
controls.panSpeed = 0.8;
controls.noZoom = false;
controls.noPan = false;
controls.staticMoving = false;
controls.dynamicDampingFactor = 0.2;

// OBJECT POSITIONS
cube.position.set(0, 0, 0);
spotlight_1.position.set(1, 1, 1);
spotlight_2.position.set(4, 3, 0);
sphere.position.set(3, 0, 0);

// ADD OBJECTS
scene.add(cube);
scene.add(spotlight_1);
scene.add(spotlight_2);
scene.add(sphere);

function animate() {

    requestAnimationFrame(animate);

    controls.update();
    //cube.rotation.x += 0.01;
    //cube.rotation.y += 0.02;

    renderer.render(scene, camera);

}
animate();
